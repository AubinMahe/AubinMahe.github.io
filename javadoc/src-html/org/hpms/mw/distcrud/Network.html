001package org.hpms.mw.distcrud;
002
003import java.io.IOException;
004import java.net.InetSocketAddress;
005import java.net.NetworkInterface;
006
007public class Network {
008
009   public static IParticipant join( int id, InetSocketAddress addr, NetworkInterface intrfc ) throws IOException {
010      return new MulticastParticipant( id, addr, intrfc );
011   }
012
013   public static IParticipant join( int id, InetSocketAddress addr ) throws IOException {
014      if( addr.getAddress().isMulticastAddress()) {
015         return new MulticastParticipant( id, addr, NetworkInterface.getByIndex( 0 ));
016      }
017      return new UDPParticipant( id, addr );
018   }
019}
